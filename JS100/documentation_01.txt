Question no 9
let tweet = 'Starting to get the hang of it... #javascript #launchschool';

tweet.split(' ');
  evaluates to>>  
  [ 'Starting',
    'to',
    'get',
    'the',
    'hang',
    'of',
    'it...',
    '#javascript',
    '#launchschool' ]

tweet.split(' ').reverse();
  evaluates to>>
  [ '#launchschool',
    '#javascript',
    'it...',
    'of',
    'hang',
    'the',
    'get',
    'to',
    'Starting' ]

tweet.split(' ').reverse().join(' ');
  evaluates to>>
  '#launchschool #javascript it... of hang the get to Starting'

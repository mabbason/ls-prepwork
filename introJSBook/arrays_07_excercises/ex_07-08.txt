Write a function similar to the oddLengths function from Exercise 6, 
but don't use map or filter. Instead, try to use the reduce method.

response:
let arr = ['a', 'abcd', 'abcde', 'abc', 'ab'];

function oddLengths(arrInput) {
  return arrInput.reduce((accArray, elmnt) => {
    if (elmnt.length % 2 === 1) {
      accArray.push(elmnt.length);
    }
    return accArray;
  }, []); 
}

console.log(oddLengths(arr)); // => [1, 5, 3]

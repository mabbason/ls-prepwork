Excercise no 6
Use map and filter to first determine the lengths of all the elements 
in an array of string values, then discard the even values 
(keep the odd values).

response:

let arr = ['a', 'abcd', 'abcde', 'abc', 'ab'];

function oddLengths(strings){
  let lengths = strings.map(element => element.length);
  let oddLengthStrings = lengths.filter(num => 
    (num % 2 === 1));
  return oddLengthStrings;
}

console.log(oddLengths(arr)); // => [1, 5, 3]

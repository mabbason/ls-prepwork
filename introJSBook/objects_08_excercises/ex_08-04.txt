Excercise no 4
Create an array from the keys of the object obj below, 
with all of the keys converted to uppercase. 
Your implementation must not mutate obj.


let obj = {
  b: 2,
  a: 1,
  c: 3,
};

response:


let obj = {
  b: 2,
  a: 1,
  c: 3,
};

let objUpperCase = Object.keys(obj);

objUpperCase = objUpperCase.map( function (element){
  return element.toUpperCase();
});

console.log(objUpperCase);
console.log(obj);
